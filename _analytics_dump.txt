class _WaterChartCard extends StatefulWidget {
  const _WaterChartCard();

  @override
  State<_WaterChartCard> createState() => _WaterChartCardState();
}

class _WaterChartCardState extends State<_WaterChartCard> {
  final _api = ApiClient();
  late final WaterApi _water = WaterApi(_api);
  List<String> _labels = const [];
  List<int> _totals = const [];
  int _target = 2000;
  bool _loading = true;

  @override
  void initState() {
    super.initState();
    _load();
  }

  Future<void> _load() async {
    try {
      final s = await _water.stats8Days();
      if (!mounted) return;
      setState(() {
        _labels = s.labels;
        _totals = s.totalsMl;
        _target = s.targetMl;
        _loading = false;
      });
    } catch (_) {
      if (mounted) setState(() => _loading = false);
    }
  }

  @override
  Widget build(BuildContext context) {
    return Container(
      padding: const EdgeInsets.all(16),
      decoration: BoxDecoration(
        color: Colors.white,
        borderRadius: BorderRadius.circular(18),
        boxShadow: const [
          BoxShadow(
            color: Colors.black12,
            blurRadius: 6,
            offset: Offset(0, 2),
          ),
        ],
      ),
      child: _loading
          ? const Center(child: CircularProgressIndicator(color: AppColors.primary))
          : Column(
              crossAxisAlignment: CrossAxisAlignment.start,
              children: [
                const Text(
                  'Р’РѕРґР° Р·Р° 8 РґРЅРµР№',
                  style: TextStyle(
                    fontFamily: 'MinecraftRus',
                    fontSize: 15,
                    fontWeight: FontWeight.bold,
                    color: AppColors.textDark,
                  ),
                ),
                const SizedBox(height: 12),
                Expanded(
                  child: BarChart(
                    BarChartData(
                      barTouchData: BarTouchData(enabled: false),
                      gridData: FlGridData(show: false),
                      borderData: FlBorderData(show: false),
                      titlesData: FlTitlesData(
                        leftTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
                        topTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
                        rightTitles: const AxisTitles(sideTitles: SideTitles(showTitles: false)),
                        bottomTitles: AxisTitles(
                          sideTitles: SideTitles(
                            showTitles: true,
                            getTitlesWidget: (value, meta) {
                              final i = value.toInt();
                              if (i < 0 || i >= _labels.length) return const SizedBox.shrink();
                              final lbl = _labels[i];
                              final short = lbl.length >= 5 ? lbl.substring(5) : lbl; // MM-dd
                              return Text(short, style: const TextStyle(fontFamily: 'MinecraftRus', fontSize: 10));
                            },
                          ),
                        ),
                      ),
                      barGroups: [
                        for (int i = 0; i < _labels.length; i++)
                          BarChartGroupData(
                            x: i,
                            barRods: [
                              BarChartRodData(
                                toY: _totals[i].toDouble(),
                                width: 14,
                                color: AppColors.primary.withOpacity(0.9),
                                borderRadius: BorderRadius.circular(6),
                                backDrawRodData: BackgroundBarChartRodData(
                                  show: true,
                                  toY: _target.toDouble(),
                                  color: Colors.grey.shade200,
                                ),
                              ),
                            ],
                          ),
                      ],
                    ),
                  ),
                ),
              ],
            ),
    );
  }
}
